1. Bar chart with div

<style>
div.bar {
  display: inline-block;
  vertical-align: bottom;
  position: relative;
  width: 20px;  /* bar width */
  background-color: cornflowerblue;
  margin-right: 2px;
}

div.text {
  position: absolute;
  bottom: 0;
  font-size: 14px;
  color: white;
  padding: 0 0 0 3px;  /* 3px padding right */
}
</style>

<div style="background-color: yellow" id="ex_2_1"></div>

<script src="./node_modules/d3/dist/d3.min.js"></script>
<script>
var dataset = [{ name: '1', value: '20px' },
               { name: '2', value: '40px' },
               { name: '3', value: '60px' }];

var bar = d3.select('#ex_2_1')
  .selectAll('div')
  .data(dataset)
  .enter()
  .append('div');

bar.style('height', d => d.value)
  .attr('class', 'bar');

bar.append('div')  //append other div to add bar labels!
  .text(d => d.name)
  .attr('class', 'text');
</script>

2. Bar chart with svg

<style>
rect.bar {
  fill: cornflowerblue;
}
text.bar {
  fill: white;
  font-size: 14px;
}
</style>

<svg width="100px" height="100px" style="background-color: yellow" id="ex_2_2"></svg>

<script src="./node_modules/d3/dist/d3.min.js"></script>
<script>
var dataset = [
  { name: '1', value: 20 },
  { name: '2', value: 40 },
  { name: '3', value: 60 }
];

var width = 20;
var height = 100;
var padding = 2;

var enter = d3.select('#ex_2_2')
  .selectAll('rect')
  .data(dataset)
  .enter();

enter.append('rect')
  .attr('x', (d, i) => i * (width + padding))
  .attr('y', d => height - d.value)
  .attr('width', d => width)
  .attr('height', d => d.value)
  .attr('class', 'bar');

enter.append('text')
  .attr('x', (d, i) => i * (width + padding) + 2) //2px from bar start
  .attr('y', d => height - padding)
  .attr('class', 'bar')
  .text(d => d.name);

enter.append("text")
  .attr('x', 50)             
  .attr('y', 10)
  .attr('text-anchor', 'middle')
  .style('font-size', '12px')
  .style('color', '#fff')
  .text('Counts');
</script>
