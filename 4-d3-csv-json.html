1. Loading data in CSV

Implement the bubble chart of ex3.1 loading the data in CSV using the sample code provided.
<script src="./node_modules/d3/dist/d3.min.js"></script>

<script>
d3.csv("cars.csv", d => { 
  return { 
    name: d.name,
    mpg: +d.mpg
  };
}).then(data => {
  d3.select("#cvs-data").text(JSON.stringify(data));
  console.log(data);
});
</script>

2. Loading data in JSON

<script src="./node_modules/d3/dist/d3.min.js"></script>
<script>
d3.json("cars.json").then((data) => {
  d3.select("#json-data").text(JSON.stringify(data));
  console.log(data);
});
</script>
